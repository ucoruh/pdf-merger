158 public static string[] SplitPdf(string inputFile, string outputFolder, int[] splitAfterPages)
159 {
160 if (string.IsNullOrEmpty(inputFile))
161 throw new ArgumentException("Input file path is required.", nameof(inputFile));
162 if (!File.Exists(inputFile))
163 throw new FileNotFoundException($"PDF file not found: {inputFile}", inputFile);
164 if (splitAfterPages == null || splitAfterPages.Length == 0)
165 throw new ArgumentException("At least one split point is required.", nameof(splitAfterPages));
166 if (string.IsNullOrEmpty(outputFolder))
167 throw new ArgumentException("Output folder path is required.", nameof(outputFolder));
168
169 using (PdfDocument inputDocument = PdfReader.Open(inputFile, PdfDocumentOpenMode.Import))
170 {
171 int pageCount = inputDocument.PageCount;
172
173 int[] sorted = splitAfterPages.Distinct().OrderBy(p => p).ToArray();
174 foreach (int p in sorted)
175 {
176 if (p < 1 || p >= pageCount)
177 throw new ArgumentException(
178 $"Split point {p} is out of range. Must be between 1 and {pageCount - 1}.",
179 nameof(splitAfterPages));
180 }
181
182 Directory.CreateDirectory(outputFolder);
183
184 List<string> outputFiles = new List<string>();
185 int startPage = 0;
186
187 for (int partIndex = 0; partIndex <= sorted.Length; partIndex++)
188 {
189 int endPage = (partIndex < sorted.Length) ? sorted[partIndex] : pageCount;
190
191 using (PdfDocument partDoc = new PdfDocument())
192 {
193 partDoc.Info.Creator = "PDF Merger";
194
195 for (int i = startPage; i < endPage; i++)
196 {
197 partDoc.AddPage(inputDocument.Pages[i]);
198 }
199
200 string outputPath = Path.Combine(outputFolder, $"split_part{partIndex + 1}.pdf");
201 partDoc.Save(outputPath);
202 outputFiles.Add(outputPath);
203 }
204
205 startPage = endPage;
206 }
207
208 return outputFiles.ToArray();
209 }

294 public static int[] ParsePageRange(string rangeString, int maxPage)
295 {
296 if (string.IsNullOrWhiteSpace(rangeString))
297 throw new ArgumentException("Page range string is required.", nameof(rangeString));
298
299 List<int> pages = new List<int>();
300 string[] parts = rangeString.Split(',');
301
302 foreach (string part in parts)
303 {
304 string trimmed = part.Trim();
305 if (string.IsNullOrEmpty(trimmed))
306 continue;
307
308 if (trimmed.Contains("-"))
309 {
310 string[] rangeParts = trimmed.Split('-');
311 if (rangeParts.Length != 2)
312 throw new ArgumentException($"Invalid range format: '{trimmed}'. Use format like '1-3'.");
313
314 int start, end;
315 if (!int.TryParse(rangeParts[0].Trim(), out start) || !int.TryParse(rangeParts[1].Trim(), out end))
316 throw new ArgumentException($"Invalid range format: '{trimmed}'. Page numbers must be integers.");
317
318 if (start > end)
319 throw new ArgumentException($"Invalid range: '{trimmed}'. Start page must be less than or equal to end page.");
320
321 if (start < 1 || end > maxPage)
322 throw new ArgumentException($"Page range '{trimmed}' is out of bounds. Valid range is 1-{maxPage}.");
323
324 for (int i = start; i <= end; i++)
325 {
326 pages.Add(i);
327 }
328 }
329 else
330 {
331 int page;
332 if (!int.TryParse(trimmed, out page))
333 throw new ArgumentException($"Invalid page number: '{trimmed}'.");
334
335 if (page < 1 || page > maxPage)
336 throw new ArgumentException($"Page number {page} is out of bounds. Valid range is 1-{maxPage}.");
337
338 pages.Add(page);
339 }
340 }
341
342 if (pages.Count == 0)
343 throw new ArgumentException("No valid page numbers found in the range string.");
344
345 return pages.ToArray();
346 }
