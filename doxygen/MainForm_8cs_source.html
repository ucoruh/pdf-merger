138 private void ExecuteMergeOperation(string operationName, Action<string, string[]> mergeAction)
139 {
140 using (SaveFileDialog dialog = new SaveFileDialog())
141 {
142 dialog.Filter = "PDF Documents|*.pdf";
143 dialog.AddExtension = true;
144 dialog.DefaultExt = "pdf";
145 DialogResult result = dialog.ShowDialog(this);
146
147 if (result != DialogResult.Cancel)
148 {
149 string[] files = new string[listFiles.Items.Count];
150 listFiles.Items.CopyTo(files, 0);
151
152 try
153 {
154 mergeAction(dialog.FileName, files);
155 Process.Start(new ProcessStartInfo(dialog.FileName) { UseShellExecute = true });
156 }
157 catch (ArgumentException ex)
158 {
159 MessageBox.Show(
160 ex.Message,
161 operationName + " - Invalid Input",
162 MessageBoxButtons.OK,
163 MessageBoxIcon.Warning);
164 }
165 catch (UnauthorizedAccessException ex)
166 {
167 MessageBox.Show(
168 "Error writing output document. You may not have write permission." +
169 Environment.NewLine + "Error: " + ex.Message,
170 operationName + " - Access Denied",
171 MessageBoxButtons.OK,
172 MessageBoxIcon.Error);
173 }
174 catch (IOException ex)
175 {
176 MessageBox.Show(
177 "Error writing output document. The file may be open in another application." +
178 Environment.NewLine + "Error: " + ex.Message,
